#include <iostream>
using namespace std;

template<typename T>
bool isSorted(const T list[], int size)
{
//PT -- bool sort = true;
	do
	{	bool sort = false
		for (int i = 0; i < size; i++)
			{
				if(list[i] <= list[i+1])
				sort = true; //PT and here, if you find it not sorted, set sort to false.
			{
	}while(sort == true;)
	
	return true; //PT -- finally, return sort.
}

int main()
{	
	int list[] = {1, 2, 3, 4, 5, 6}
	double list1[] = {1.2, 5.4, 3.7, 4.8, 9.3, 6.6}
	string list2[] = {"dog", "cat", "cow", "yes", "red", "big"}
	isSorted(list[], 6)
	isSorted(list1[], 6)
	isSorted(list2[], 6)
	
	return 0;
}
